# ── Core ─────────────────────────────────
set $mod Mod4
set $term kitty
set $menu wofi --show drun --style ~/.config/wofi/style.css

# ── Output / Wallpaper ────────────────────

# ── Gaps & Borders ───────────────────────
gaps inner 8
gaps outer 12
default_border normal 2
default_floating_border normal 2
smart_gaps on
smart_borders on

# ── Colors (black/yellow) ────────────────
# class                 border    bg        text      indicator child_border
client.focused          #F5C400   #0A0A0A   #F5C400   #F5C400   #F5C400

# ── Font ─────────────────────────────────
font pango:DejaVu Sans Mono 10

# ── Input ────────────────────────────────
input "type:keyboard" {
    xkb_options caps:escape
}
input "type:touchpad" {
    tap enabled
    natural_scroll enabled
    dwt enabled
}

# ── Seat ─────────────────────────────────
seat seat0 xcursor_theme Adwaita 24

# ── Key Bindings ─────────────────────────

# Terminal
bindsym $mod+Return exec $term

# Launcher
bindsym $mod+d exec $menu

bindsym $mod+m exec python3 ~/.config/militechos/mission-menu.py

# Kill focused window
bindsym $mod+Shift+q kill

# Reload / Restart
bindsym $mod+Shift+r reload

# Exit sway
bindsym $mod+Shift+e exec swaynag -t warning \
    -m 'EXIT MilitechOS?' \
    -b 'CONFIRM' 'swaymsg exit'

# Screenshots
bindsym Print exec grim ~/Pictures/screenshot-$(date +%Y%m%d-%H%M%S).png
bindsym $mod+Print exec grim -g "$(slurp)" ~/Pictures/screenshot-$(date +%Y%m%d-%H%M%S).png

# Volume
bindsym XF86AudioRaiseVolume exec pactl set-sink-volume @DEFAULT_SINK@ +5%
bindsym XF86AudioLowerVolume exec pactl set-sink-volume @DEFAULT_SINK@ -5%
bindsym XF86AudioMute exec pactl set-sink-mute @DEFAULT_SINK@ toggle

# Brightness (install brightnessctl if needed)
bindsym XF86MonBrightnessUp exec brightnessctl set +10%
bindsym XF86MonBrightnessDown exec brightnessctl set 10%-

# Focus movement (vim keys + arrows)
bindsym $mod+h focus left

bindsym $mod+k focus up

bindsym $mod+Up focus up
bindsym $mod+Right focus right

# Move windows
bindsym $mod+Shift+h move left
bindsym $mod+Shift+j move down
bindsym $mod+Shift+k move up
bindsym $mod+Shift+l move right
bindsym $mod+Shift+Left move left
bindsym $mod+Shift+Down move down
bindsym $mod+Shift+Up move up
bindsym $mod+Shift+Right move right

# Splits
bindsym $mod+b splith
bindsym $mod+v splitv

# Layout toggle
bindsym $mod+s layout stacking
bindsym $mod+w layout tabbed
bindsym $mod+e layout toggle split

# Fullscreen
bindsym $mod+f fullscreen toggle

# Floating toggle
bindsym $mod+Shift+space floating toggle
bindsym $mod+space focus mode_toggle

# Sticky (float on all workspaces)
bindsym $mod+Shift+s sticky toggle

# Thunar File Manager
bindsym $mod+t exec thunar

# ── Resize Mode ──────────────────────────
mode "RESIZE" {
    bindsym h resize shrink width 20px
    bindsym j resize grow height 20px
    bindsym k resize shrink height 20px
    bindsym l resize grow width 20px
    bindsym Left resize shrink width 20px
    bindsym Down resize grow height 20px
    bindsym Up resize shrink height 20px
    bindsym Right resize grow width 20px
    bindsym Return mode "default"
    bindsym Escape mode "default"
}
bindsym $mod+r mode "RESIZE"
